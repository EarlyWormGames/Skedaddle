65  public static Matrix4x4 GetPerspectiveProjection(float left, float right, float bottom, float top, float near, float far)
66  {
67  float x = (2.0f * near) / (right - left);
68  float y = (2.0f * near) / (top - bottom);
69  float a = (right + left) / (right - left);
70  float b = (top + bottom) / (top - bottom);
71  float c = -(far + near) / (far - near);
72  float d = -(2.0f * far * near) / (far - near);
73  float e = -1.0f;
74 
75  Matrix4x4 m = new Matrix4x4();
76  m[0, 0] = x; m[0, 1] = 0; m[0, 2] = a; m[0, 3] = 0;
77  m[1, 0] = 0; m[1, 1] = y; m[1, 2] = b; m[1, 3] = 0;
78  m[2, 0] = 0; m[2, 1] = 0; m[2, 2] = c; m[2, 3] = d;
79  m[3, 0] = 0; m[3, 1] = 0; m[3, 2] = e; m[3, 3] = 0;
80  return m;

91  public static Matrix4x4 GetPerspectiveProjection(this Camera camera, float texelOffsetX, float texelOffsetY)
92  {
93  if (camera == null)
94  return Matrix4x4.identity;
95 
96  float oneExtentY = Mathf.Tan(0.5f * Mathf.Deg2Rad * camera.fieldOfView);
97  float oneExtentX = oneExtentY * camera.aspect;
98  float texelSizeX = oneExtentX / (0.5f * camera.pixelWidth);
99  float texelSizeY = oneExtentY / (0.5f * camera.pixelHeight);
100  float oneJitterX = texelSizeX * texelOffsetX;
101  float oneJitterY = texelSizeY * texelOffsetY;
102 
103  float cf = camera.farClipPlane;
104  float cn = camera.nearClipPlane;
105  float xm = (oneJitterX - oneExtentX) * cn;
106  float xp = (oneJitterX + oneExtentX) * cn;
107  float ym = (oneJitterY - oneExtentY) * cn;
108  float yp = (oneJitterY + oneExtentY) * cn;
109 
110  return Matrix4x4Extension.GetPerspectiveProjection(xm, xp, ym, yp, cn, cf);
111  }
