443  // Apply the curve to the specified solver, effector, with the value and weight.
444  private void Apply(IKSolverFullBodyBiped solver, FullBodyBipedEffector effector, WeightCurve.Type type, float value, float weight) {
445  switch(type) {
446  case WeightCurve.Type.PositionWeight:
447  solver.GetEffector(effector).positionWeight = Mathf.Lerp(solver.GetEffector(effector).positionWeight, value, weight);
448  return;
449  case WeightCurve.Type.RotationWeight:
450  solver.GetEffector(effector).rotationWeight = Mathf.Lerp(solver.GetEffector(effector).rotationWeight, value, weight);
451  return;
452  case WeightCurve.Type.PositionOffsetX:
453  solver.GetEffector(effector).position += (positionOffsetSpace != null? positionOffsetSpace.rotation: solver.GetRoot().rotation) * Vector3.right * value * weight;
454  return;
455  case WeightCurve.Type.PositionOffsetY:
456  solver.GetEffector(effector).position += (positionOffsetSpace != null? positionOffsetSpace.rotation: solver.GetRoot().rotation) * Vector3.up * value * weight;
457  return;
458  case WeightCurve.Type.PositionOffsetZ:
459  solver.GetEffector(effector).position += (positionOffsetSpace != null? positionOffsetSpace.rotation: solver.GetRoot().rotation) * Vector3.forward * value * weight;
460  return;
461  case WeightCurve.Type.Pull:
462  solver.GetChain(effector).pull = Mathf.Lerp(solver.GetChain(effector).pull, value, weight);
463  return;
464  case WeightCurve.Type.Reach:
465  solver.GetChain(effector).reach = Mathf.Lerp(solver.GetChain(effector).reach, value, weight);
466  return;
467  case WeightCurve.Type.Push:
468  solver.GetChain(effector).push = Mathf.Lerp(solver.GetChain(effector).push, value, weight);
469  return;
470  case WeightCurve.Type.PushParent:
471  solver.GetChain(effector).pushParent = Mathf.Lerp(solver.GetChain(effector).pushParent, value, weight);
472  return;
473  }
474  }
