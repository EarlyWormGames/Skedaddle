88  // Is the character in range with this character position?
89  public bool IsInRange(Transform character, Transform trigger, out float error) {
90  // Do not use this character position, trigger is still valid
91  error = 0f;
92  if (!use) return true;
93 
94  // Invalid character position conditions
95  error = 180f;
96  if (radius <= 0f) return false;
97  if (maxAngle <= 0f) return false;
98 
99  Vector3 forward = trigger.forward;
100  if (fixYAxis) forward.y = 0f;
101  if (forward == Vector3.zero) return false; // Singularity
102 
103  Vector3 up = (fixYAxis? Vector3.up: trigger.up);
104 
105  Quaternion triggerRotation = Quaternion.LookRotation(forward, up);
106 
107  Vector3 position = trigger.position + triggerRotation * offset3D;
108 
109  Vector3 origin = orbit? trigger.position: position;
110  Vector3 toCharacter = character.position - origin;
111  Vector3.OrthoNormalize(ref up, ref toCharacter);
112  toCharacter *= Vector3.Project(character.position - origin, toCharacter).magnitude;
113 
114  if (orbit) {
115  float mag = offset.magnitude;
116  float dist = toCharacter.magnitude;
117  if (dist < mag - radius || dist > mag + radius) return false;
118  } else {
119  if (toCharacter.magnitude > radius) return false;
120  }
121 
122  Vector3 d = triggerRotation * direction3D;
123  Vector3.OrthoNormalize(ref up, ref d);
124 
125  if (orbit) {
126  Vector3 toPosition = position - trigger.position;
127  if (toPosition == Vector3.zero) toPosition = Vector3.forward;
128  Quaternion r = Quaternion.LookRotation(toPosition, up);
129  toCharacter = Quaternion.Inverse(r) * toCharacter;
130 
131  float a = Mathf.Atan2(toCharacter.x, toCharacter.z) * Mathf.Rad2Deg;
132  d = Quaternion.AngleAxis(a, up) * d;
133  }
134 
135  float angle = Vector3.Angle(d, character.forward);
136  if (angle > maxAngle) return false;
137  error = (angle / maxAngle) * 180f;
138 
139  return true;
140  }
