417  /*
418  * Tries to guess which bone should be the root node
419  * */
420  public static Transform DetectRootNodeBone(BipedReferences references) {
421  if (!references.isFilled) return null;
422  if (references.spine.Length < 1) return null;
423 
424  int spineLength = references.spine.Length;
425  if (spineLength == 1) return references.spine[0];
426 
427  Vector3 hip = Vector3.Lerp(references.leftThigh.position, references.rightThigh.position, 0.5f);
428  Vector3 neck = Vector3.Lerp(references.leftUpperArm.position, references.rightUpperArm.position, 0.5f);
429  Vector3 toNeck = neck - hip;
430  float toNeckMag = toNeck.magnitude;
431 
432  if (references.spine.Length < 2) return references.spine[0];
433 
434  int rootNodeBone = 0;
435 
436  for (int i = 1; i < spineLength; i++) {
437  Vector3 hipToBone = references.spine[i].position - hip;
438  Vector3 projection = Vector3.Project(hipToBone, toNeck);
439 
440  float dot = Vector3.Dot(projection.normalized, toNeck.normalized);
441  if (dot > 0) {
442  float mag = projection.magnitude / toNeckMag;
443  if (mag < 0.5f) rootNodeBone = i;
444  }
445  }
446 
447  return references.spine[rootNodeBone];
448  }
