182  private Vector3 GetModifiedBendNormal() {
183  float weight = bendModifierWeight;
184  if (weight <= 0) return bendNormal;
185 
186  switch(bendModifier) {
187  // Animation Bend Mode attempts to maintain the bend axis as it is in the animation
188  case BendModifier.Animation:
189  if (!maintainBendFor1Frame) MaintainBend();
190  maintainBendFor1Frame = false;
191  return Vector3.Lerp(bendNormal, animationNormal, weight);
192 
193  // Bending relative to the parent of the first bone
194  case BendModifier.Parent:
195  if (bone1.transform.parent == null) return bendNormal;
196 
197  Quaternion parentRotation = bone1.transform.parent.rotation * Quaternion.Inverse(parentDefaultRotation);
198  return Quaternion.Slerp(Quaternion.identity, parentRotation * Quaternion.Inverse(defaultRootRotation), weight) * bendNormal;
199 
200  // Bending relative to IKRotation
201  case BendModifier.Target:
202  Quaternion targetRotation = IKRotation * Quaternion.Inverse(bone3DefaultRotation);
203  return Quaternion.Slerp(Quaternion.identity, targetRotation, weight) * bendNormal;
204 
205  // Anatomic Arm
206  case BendModifier.Arm:
207  if (bone1.transform.parent == null) return bendNormal;
208 
209  // Disabling this for legs
210  if (goal == AvatarIKGoal.LeftFoot || goal == AvatarIKGoal.RightFoot) {
211  if (!Warning.logged) LogWarning("Trying to use the 'Arm' bend modifier on a leg.");
212  return bendNormal;
213  }
214 
215  Vector3 direction = (IKPosition - bone1.transform.position).normalized;
216 
217  // Converting direction to default world space
218  direction = Quaternion.Inverse(bone1.transform.parent.rotation * Quaternion.Inverse(parentDefaultRotation)) * direction;
219 
220  // Inverting direction for left hand
221  if (goal == AvatarIKGoal.LeftHand) direction.x = -direction.x;
222 
223  // Calculating dot products for all AxisDirections
224  for (int i = 1; i < axisDirections.Length; i++) {
225  axisDirections[i].dot = Mathf.Clamp(Vector3.Dot(axisDirections[i].direction, direction), 0f, 1f);
226  axisDirections[i].dot = Interp.Float(axisDirections[i].dot, InterpolationMode.InOutQuintic);
227  }
228 
229  // Summing up the arm bend axis

231

237  sum.x = -sum.x;
238  sum = -sum;
239  }
240 
241  // Converting sum back to parent space
242  Vector3 armBendNormal = bone1.transform.parent.rotation * Quaternion.Inverse(parentDefaultRotation) * sum;
243 
244  if (weight >= 1) return armBendNormal;
245  return Vector3.Lerp(bendNormal, armBendNormal, weight);
246 
247  // Bending towards the bend goal Transform
248  case BendModifier.Goal:
249  if (bendGoal == null) {
