186  public RaycastHit GetRootHit() {
187  RaycastHit h = new RaycastHit();
188  Vector3 _up = up;
189 
190  Vector3 legsCenter = Vector3.zero;
191  foreach (Leg leg in legs) legsCenter += leg.transform.position;
192  legsCenter /= legs.Length;
193 
194  h.point = legsCenter - _up * maxStep * 10f;
195  float distMlp = 10 + 1;
196  h.distance = maxStep * distMlp;
197 
198  if (maxStep <= 0f) return h;
199 
200  if (quality != Quality.Best) Physics.Raycast(legsCenter + _up * maxStep, -_up, out h, maxStep * distMlp, layers);
201  else Physics.SphereCast(legsCenter + _up * maxStep, rootSphereCastRadius, -up, out h, maxStep * distMlp, layers);
202
