141  private static int DrawLayerMaskField(GUIContent content, int layerMask)
142  {
143  string[] layerNames = UnityEditorInternal.InternalEditorUtility.layers;
144  LayerNumbers.Clear();
145 
146  for (int i = 0; i < layerNames.Length; i++)

150 
151  int maskWithoutEmpty = 0;
152  for (int i = 0; i < LayerNumbers.Count; i++)
153  {
154  if (((1 << LayerNumbers[i]) & layerMask) > 0)
155  maskWithoutEmpty |= (1 << i);

157 
158  maskWithoutEmpty = EditorGUILayout.MaskField(content, maskWithoutEmpty, layerNames);
159 
160  int mask = 0;
