297  //Calculates the bend direction based on the law of cosines. NB! Magnitude of the returned vector does not equal to the length of the first bone!
298  private static Vector3 GetDirectionToBendPoint(Vector3 direction, float directionMag, Vector3 bendDirection, float sqrMag1, float sqrMag2) {
299  float x = ((directionMag * directionMag) + (sqrMag1 - sqrMag2)) / 2f / directionMag;
300  float y = (float)Math.Sqrt(Mathf.Clamp(sqrMag1 - x * x, 0, Mathf.Infinity));
301 
302  if (direction == Vector3.zero) return Vector3.zero;
303  return Quaternion.LookRotation(direction, bendDirection) * new Vector3(0f, y, x);
304  }
