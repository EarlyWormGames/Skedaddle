203  private Vector3[] SmoothPoints() {
204  // Create the new point array with double length
205  Vector3[] Q = new Vector3[P.Length * 2];
206 
207  float scalar = GetScalar(P.Length); // Get the constant used for interpolation
208 
209  // Project all the existing points on a plane that is tangent to the unit sphere at the Axis point
210  for (int i = 0; i < Q.Length; i+= 2) Q[i] = PointToTangentPlane(P[i / 2], 1);
211 
212  // Interpolate the new points
213  for (int i = 1; i < Q.Length; i+= 2) {
214  Vector3 minus2 = Vector3.zero;
215  Vector3 plus1 = Vector3.zero;
216  Vector3 plus2 = Vector3.zero;
217 
218  if (i > 1 && i < Q.Length - 2) {

220  plus2 = Q[i + 1];

224  } else if (i == Q.Length - 1) {

280  * Applies Swing limit to the rotation
281  * */
282  private Quaternion LimitSwing(Quaternion rotation) {
283  if (rotation == Quaternion.identity) return rotation; // Assuming initial rotation is in the reachable area
284 
285  Vector3 L = rotation * axis; // Test this vector against the reach cones
286 
287  int r = GetReachCone(L); // Get the reach cone to test against (can be only 1)
288 
289  // Just in case we are running our application with invalid reach cones
290  if (r == -1) {
291  if (!Warning.logged) LogWarning("RotationLimitPolygonal reach cones are invalid.");
292  return rotation;
293  }
294 
295  // Dot product of cone normal and rotated axis
296  float v = Vector3.Dot(reachCones[r].B, L);
297  if (v > 0) return rotation; // Rotation is reachable
298 
299  // Find normal for a plane defined by origin, axis, and rotated axis
300  Vector3 rotationNormal = Vector3.Cross(axis, L);
301 
302  // Find the line where this plane intersects with the reach cone plane
303  L = Vector3.Cross(-reachCones[r].B, rotationNormal);
304 
305  // Rotation from current(illegal) swing rotation to the limited(legal) swing rotation
306  Quaternion toLimits = Quaternion.FromToRotation(rotation * axis, L);
307 
308  // Subtract the illegal rotation
309  return toLimits * rotation;
310  }
