95  private static void TransformPattern(float[] seq, float theta, float scale)
96  {
97  float cs = Mathf.Cos(theta);
98  float sn = Mathf.Sin(theta);
99  for (int i = 0, j = 1, n = seq.Length; i != n; i += 2, j += 2)
100  {
101  float x = scale * seq[i];
102  float y = scale * seq[j];
103  seq[i] = x * cs - y * sn;
104  seq[j] = x * sn + y * cs;
105  }
106  }

108  // http://en.wikipedia.org/wiki/Halton_sequence
109  private static float HaltonSeq(int prime, int index = 1/* NOT! zero-based */)
110  {
111  float r = 0f;
112  float f = 1f;
113  int i = index;
114  while (i > 0)
115  {
116  f /= prime;
117  r += f * (i % prime);
118  i = (int)Mathf.Floor(i / (float)prime);
119  }
120  return r;
121  }

123  private static void InitializeHalton_2_3(float[] seq)
124  {
125  for (int i = 0, n = seq.Length / 2; i != n; i++)

128  float v = HaltonSeq(3, i + 1) - 0.5f;
129  seq[2 * i + 0] = u;

131  }

231  public Vector2 Sample(Pattern pattern, int index)
232  {
233  float[] points = AccessPointData(pattern);
234  int n = points.Length / 2;
235  int i = index % n;
236 
237  float x = patternScale * points[2 * i + 0];
238  float y = patternScale * points[2 * i + 1];
239 
240  if (pattern != Pattern.MotionPerp2)
241  return new Vector2(x, y);
242  else
243  return new Vector2(x, y).Rotate(Vector2.right.SignedAngle(focalMotionDir));
244  }

246  void OnPreCull()
247  {
248  var camera = GetComponent<Camera>();
249  if (camera != null && camera.orthographic == false)
250  {
251  // update motion dir
252  {
253  Vector3 oldWorld = focalMotionPos;
254  Vector3 newWorld = camera.transform.TransformVector(camera.nearClipPlane * Vector3.forward);
255 
256  Vector3 oldPoint = (camera.worldToCameraMatrix * oldWorld);
257  Vector3 newPoint = (camera.worldToCameraMatrix * newWorld);
258  Vector3 newDelta = (newPoint - oldPoint).WithZ(0f);
259 
260  var mag = newDelta.magnitude;
261  if (mag != 0f)
262  {
263  var dir = newDelta / mag;// yes, apparently this is necessary instead of newDelta.normalized... because facepalm
264  if (dir.sqrMagnitude != 0f)
265  {
266  focalMotionPos = newWorld;
267  focalMotionDir = Vector3.Slerp(focalMotionDir, dir, 0.2f);
268  //Debug.Log("CHANGE focalMotionDir " + focalMotionDir.ToString("G4") + " delta was " + newDelta.ToString("G4") + " delta.mag " + newDelta.magnitude);
269  }
270  }
271  }
272 
273  // update jitter
274  {
275  activeIndex += 1;
276  activeIndex %= AccessLength(pattern);
277 
278  Vector2 sample = Sample(pattern, activeIndex);
279  activeSample.z = activeSample.x;
280  activeSample.w = activeSample.y;
281  activeSample.x = sample.x;
282  activeSample.y = sample.y;
283 
284  camera.projectionMatrix = camera.GetPerspectiveProjection(sample.x, sample.y);
285  }
286  }

288  {
289  activeSample = Vector4.zero;
290  activeIndex = -1;
291  }
