60  void Resolve(RenderTexture source, RenderTexture destination)
61  {
62  EnsureMaterial(ref reprojectionMaterial, reprojectionShader);
63 
64  if (_camera.orthographic || _camera.depthTextureMode == DepthTextureMode.None || reprojectionMaterial == null)
65  {
66  Graphics.Blit(source, destination);
67  if (_camera.depthTextureMode == DepthTextureMode.None)
68  _camera.depthTextureMode = DepthTextureMode.Depth;
69  return;
70  }
71 
72  if (reprojectionMatrix == null || reprojectionMatrix.Length != 2)
73  reprojectionMatrix = new Matrix4x4[2];
74  if (reprojectionBuffer == null || reprojectionBuffer.Length != 2)
75  reprojectionBuffer = new RenderTexture[2];
76 
77  int bufferW = source.width;
78  int bufferH = source.height;
79 
80  EnsureRenderTarget(ref reprojectionBuffer[0], bufferW, bufferH, RenderTextureFormat.ARGB32, FilterMode.Bilinear);
81  EnsureRenderTarget(ref reprojectionBuffer[1], bufferW, bufferH, RenderTextureFormat.ARGB32, FilterMode.Bilinear);
82 
83  EnsureKeyword(reprojectionMaterial, "MINMAX_3X3", neighborhood == Neighborhood.MinMax3x3);
84  EnsureKeyword(reprojectionMaterial, "MINMAX_3X3_ROUNDED", neighborhood == Neighborhood.MinMax3x3Rounded);
85  EnsureKeyword(reprojectionMaterial, "MINMAX_4TAP_VARYING", neighborhood == Neighborhood.MinMax4TapVarying);
86  EnsureKeyword(reprojectionMaterial, "UNJITTER_COLORSAMPLES", unjitterColorSamples);
87  EnsureKeyword(reprojectionMaterial, "UNJITTER_NEIGHBORHOOD", unjitterNeighborhood);
88  EnsureKeyword(reprojectionMaterial, "UNJITTER_REPROJECTION", unjitterReprojection);
89  EnsureKeyword(reprojectionMaterial, "USE_YCOCG", useYCoCg);
90  EnsureKeyword(reprojectionMaterial, "USE_CLIPPING", useClipping);
91  EnsureKeyword(reprojectionMaterial, "USE_DILATION", useDilation);
92 #if UNITY_EDITOR
93  EnsureKeyword(reprojectionMaterial, "USE_MOTION_BLUR", Application.isPlaying ? useMotionBlur : false);
94 #else
95  EnsureKeyword(reprojectionMaterial, "USE_MOTION_BLUR", useMotionBlur);
96 #endif
97  EnsureKeyword(reprojectionMaterial, "USE_MOTION_BLUR_NEIGHBORMAX", _velocity.velocityNeighborMax != null);
98  EnsureKeyword(reprojectionMaterial, "USE_OPTIMIZATIONS", useOptimizations);
99 
100  Matrix4x4 cameraP = _camera.GetPerspectiveProjection();
101  Matrix4x4 cameraVP = cameraP * _camera.worldToCameraMatrix;
102 
103  float oneExtentY = Mathf.Tan(0.5f * Mathf.Deg2Rad * _camera.fieldOfView);
104  float oneExtentX = oneExtentY * _camera.aspect;
105 
106  if (reprojectionIndex == -1)// bootstrap
107  {
108  reprojectionIndex = 0;
109  reprojectionMatrix[reprojectionIndex] = cameraVP;
110  Graphics.Blit(source, reprojectionBuffer[reprojectionIndex]);
111  }
112 
113  int indexRead = reprojectionIndex;
114  int indexWrite = (reprojectionIndex + 1) % 2;
115 
116  reprojectionMaterial.SetTexture("_VelocityBuffer", _velocity.velocityBuffer);
117  reprojectionMaterial.SetTexture("_VelocityNeighborMax", _velocity.velocityNeighborMax);
118  reprojectionMaterial.SetVector("_Corner", new Vector4(oneExtentX, oneExtentY, 0f, 0f));
119  reprojectionMaterial.SetVector("_Jitter", _jitter.activeSample);
120  reprojectionMaterial.SetMatrix("_PrevVP", reprojectionMatrix[indexRead]);
121  reprojectionMaterial.SetTexture("_MainTex", source);
122  reprojectionMaterial.SetTexture("_PrevTex", reprojectionBuffer[indexRead]);
123  reprojectionMaterial.SetFloat("_FeedbackMin", feedbackMin);
124  reprojectionMaterial.SetFloat("_FeedbackMax", feedbackMax);
125  reprojectionMaterial.SetFloat("_MotionScale", motionBlurStrength * (motionBlurIgnoreFF ? Mathf.Min(1f, 1f / _velocity.timeScale) : 1f));
126 
127  // reproject frame n-1 into output + history buffer
128  {
129  mrt[0] = reprojectionBuffer[indexWrite].colorBuffer;
130  mrt[1] = destination.colorBuffer;
131 
132  Graphics.SetRenderTarget(mrt, source.depthBuffer);
133  reprojectionMaterial.SetPass(0);
134 
135  FullScreenQuad();
136 
137  reprojectionMatrix[indexWrite] = cameraVP;
138  reprojectionIndex = indexWrite;
139  }
140  }
