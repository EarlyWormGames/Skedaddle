78  private static void AddLimbHelper(FBIKChain chain, float size, Transform root = null) {
79  Vector3 cross = Vector3.Cross((chain.nodes[1].transform.position - chain.nodes[0].transform.position).normalized, (chain.nodes[2].transform.position - chain.nodes[0].transform.position).normalized);
80 
81  Vector3 bendDirection = -Vector3.Cross(cross.normalized, (chain.nodes[2].transform.position - chain.nodes[0].transform.position).normalized);
82 
83  if (bendDirection != Vector3.zero) {
84  Color c = Handles.color;
85  bool inverted = root != null && Vector3.Dot(root.forward, bendDirection.normalized) < 0f;
86 
87  // Inverted bend direction
88  if (inverted) {
89  GUI.color = new Color(1f, 0.75f, 0.75f);
90  Handles.color = Color.yellow;
91 
92  if (Handles.Button(chain.nodes[1].transform.position, Quaternion.identity, size * 0.5f, size, Handles.DotCap)) {
93  Warning.logged = false;
94  Warning.Log("The bend direction of this limb appears to be inverted. Please rotate this bone so that the limb is bent in it's natural bending direction. If this limb is supposed to be bent in the direction pointed by the arrow, ignore this warning.", root, true);
95  }
96  }
97 
98  Handles.ArrowCap(0, chain.nodes[1].transform.position, Quaternion.LookRotation(bendDirection), size * 2f);
99 
100  GUI.color = Color.white;
101  Handles.color = c;
102  } else {
103  // The limb is completely stretched out
104  Color c = Handles.color;
105  Handles.color = Color.red;
106  GUI.color = new Color(1f, 0.75f, 0.75f);
107 
108  if (Handles.Button(chain.nodes[1].transform.position, Quaternion.identity, size * 0.5f, size, Handles.DotCap)) {
109  Warning.logged = false;
110  Warning.Log("The limb is completely stretched out. Full Body Biped IK does not know which way the limb should be bent. Please rotate this bone slightly in it's bending direction.", root, true);
111  }
112 
113  GUI.color = Color.white;
114  Handles.color = c;
